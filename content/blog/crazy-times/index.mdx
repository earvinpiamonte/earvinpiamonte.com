function generateStarPattern(size = 1) {
  const STAR = '*';
  const SPACE = ' ';
  const NEW_LINE = '\n';

  let stars = '';

  if (size >= 10) {
    return 'ðŸŒŒ';
  }

  for (let row = 1; row <= size; row++) {
    let paddingSize = size - row;
    let paddingCharacters = `${SPACE}`.repeat(paddingSize);
    let cellStar = `${STAR}${SPACE}`.repeat(row).trim();
    let rowStars = `${paddingCharacters}${cellStar}${NEW_LINE}`;

    stars += rowStars;
  }

  return stars;
}
